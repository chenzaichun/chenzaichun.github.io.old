#!/usr/bin/env python

from sympy import *

var('x')
print(solve(log(x-1,2)+log(x,2)-1,x))

结果是2，可是貌似还有一个负数解呢( -1 哪儿去了?)

#!/usr/bin/env python

from sympy import *

var('x')
print(solve(log(x-1,2)+log(x,2)-1,x))

y = Symbol('y', negative=True)
print(solve(log(y-1,2)+log(y,2)-1,y))

暂时只能通过这种方式求解了。
